task job уровень контрольной памяти
```bash
#!/bin/bash
#SBATCH --nodes=1
#SBATCH --ntasks-per-node=2
#SBATCH --job-name=Oak_Level1_Memory
#SBATCH --output=Oak_level1_%j.out

module load mpi/mvapich2-2.2  # или проверьте доступные модули: `module avail`

cd $SLURM_SUBMIT_DIR

# Оба процесса на сокете 0 (локальная память)
numactl --cpunodebind=0 --membind=0 \
  mpiexec -np 2 ./main > oak_level1_memory.txt
```

task job Шина qpi
```bash
#!/bin/bash
#SBATCH --nodes=1
#SBATCH --ntasks-per-node=2
#SBATCH --job-name=Oak_Level2_QPI
#SBATCH --output=Oak_level2_%j.out

module load mpi/mvapich2-2.2

cd $SLURM_SUBMIT_DIR

# Распределение по разным NUMA-узлам (сокетам)
mpiexec -np 2 --map-by ppr:1:numa --bind-to core \
  ./main > oak_level2_qpi.txt
```
